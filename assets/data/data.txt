function showTime() {
    let time = new Date();
    let hour = time.getHours();
    let min = time.getMinutes();
    let sec = time.getSeconds();
    
    hour = hour < 10 ? "0" + hour : hour;
    min = min < 10 ? "0" + min : min;
    sec = sec < 10 ? "0" + sec : sec;

    let currentTime = hour + ":" +
        min + ":" + sec;

    document.getElementById("jam_digital")
        .innerHTML = currentTime;
}

var convert = {
    toCompass: function (degrees) {
        if (degrees) {
            return ['Utara', 'Utara-Timur Laut', 'Timur Laut', 'Timur-Timur Laut', 'Timur', 'Timur-Tenggara', 'Tenggara', 'Selatan-Tenggara', 'Selatan', 'Selatan-Barat Daya',
                'Barat Daya', 'Barat-Barat Daya', 'Barat', 'Barat-Barat Laut', 'Barat Laut', 'Utara-Barat Laut', 'Utara'
            ][Math.round(degrees / 11.25 / 2)];

        } else {
            return "-"
        }
    }
}
//logegr panah kompas
function rotateCompas(x) {
    if (x) {
        return [0, 22.5, 45, 67.5, 90, 112.5, 135, 157.5, 180, 202.5,
            225, 247.5, 270, 292.5, 315, 337.5, 360
        ][Math.round(x / 11.25 / 2)];

    } else {
        return "-"
    }
}

function checkIfImageExists(url, callback) {
    const img = new Image();
    img.src = url;
    
    if (img.complete) {
      callback(true);
    } else {
      img.onload = () => {
        callback(true);
      };
      
      img.onerror = () => {
        callback(false);
      };
    }
  }
